export default {
  async fetch(request) {

    const url = new URL(request.url);
    const target = url.searchParams.get('url');

    if (!target) {
      return new Response('url param required', {
        status: 400,
        headers: corsHeaders('text/plain; charset=UTF-8')
      });
    }

    if (request.method === 'OPTIONS') {
      return new Response(null, {
        status: 204,
        headers: corsHeaders('text/plain; charset=UTF-8')
      });
    }

    const res = await fetch(target, {
      method: 'GET',
      headers: {
        'User-Agent':
          'Mozilla/5.0 (iPhone; CPU iPhone OS 17_0 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) Version/17.0 Mobile Safari/605.1',

        'Accept':
          'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',

        'Accept-Language':
          'ja,en-US;q=0.9,en;q=0.8',

        'Referer':
          'https://race.netkeiba.com/',

        'Cache-Control':
          'no-cache',

        'Pragma':
          'no-cache'
      }
    });

    const html = await res.text();

    return new Response(html, {
      status: res.status,
      headers: corsHeaders('text/html; charset=UTF-8')
    });
  }
};

function corsHeaders(contentType) {
  return {
    'Content-Type': contentType,
    'Access-Control-Allow-Origin': '*',
    'Access-Control-Allow-Methods': 'GET, OPTIONS',
    'Access-Control-Allow-Headers': '*',
    'Cache-Control': 'no-store'
  };
}
